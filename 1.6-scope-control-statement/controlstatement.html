If Else and Switch case, decision based program flow

<script>
let age = 18;

if (age > 18) {
//debugger;
console.log("Allowed for driving");
} else if (age === 18) {//=== checks the value and also the data type for doing comparision, but == only checks the value
console.log("You can now apply for driving license");
}
else {
console.log("Not allowed for driving");
}
const day = 3;
switch (day) {
case 1:
console.log("Today is Monday");
break;

case 3:
console.log("Today is Wednesday");
break;
case 6:
console.log("Today is Saturday");
break;

default:
console.log("I dont know what day it is today");
break;
}
</script>
