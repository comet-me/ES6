function deleteCart() {
//confirm will help user to decide on an action
let input = confirm("Are you sure you want to delete the cart?");
if (input) {
alert("Cart has been deleted successfully!");
} else {
alert("Cart cannot be deleted!");
}
}
deleteCart();
